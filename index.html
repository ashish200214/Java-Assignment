Q3. Write java program for demonstrating concept of Thread synchronization.

package mypack;
import java.io.*;
import java.util.*;

// A Class used to send a message
class Sender {
	public void send(String msg)
	{
		System.out.println("Sending\t" + msg);
		try {
			Thread.sleep(1000);
		}
		catch (Exception e) {
			System.out.println("Thread interrupted.");
		}
		System.out.println("\n" + msg + "Sent");
	}
}

// Class for send a message using Threads
class ThreadedSend extends Thread {
	private String msg;
	Sender sender;

	// Receives a message object and a string
	// message to be sent
	ThreadedSend(String m, Sender obj)
	{
		msg = m;
		sender = obj;
	}

	public void run()
	{
		// Only one thread can send a message
		// at a time.
		synchronized (sender)
		{
			// synchronizing the send object
			sender.send(msg);
		}
	}
}

// Driver class
class Q3 {
	public static void main(String args[])
	{
		Sender send = new Sender();
		ThreadedSend S1 = new ThreadedSend(" Hi ", send);
		ThreadedSend S2 = new ThreadedSend(" Bye ", send);

		// Start two threads of ThreadedSend type
		S1.start();
		S2.start();

		// wait for threads to end
		try {
			S1.join();
			S2.join();
		}
		catch (Exception e) {
			System.out.println("Interrupted");
		}
	}
}

Q4. Write java code for implementing the following Inter-thread communication methods:
usingwait(), notify(), notifyAll()

package mypack;
class Data {
    private int number;
    private boolean isValueSet = false;

    public synchronized void produce(int num) {
        // Wait until consumer consumes the value
        while (isValueSet) {
            try {
                wait();
            } catch (InterruptedException e) {
                e.printStackTrace();
            }
        }
        number = num;
        System.out.println("Produced: " + num);
        isValueSet = true;
        // Notify the consumer that value is ready
        notify();
    }

    public synchronized void consume() {
        // Wait until producer produces the value
        while (!isValueSet) {
            try {
                wait();
            } catch (InterruptedException e) {
                e.printStackTrace();
            }
        }
        System.out.println("Consumed: " + number);
        isValueSet = false;
        // Notify the producer that value is consumed
        notify();
    }
}

public class Q4 {
    public static void main(String[] args) {
        final Data data = new Data();

        // Producer thread
        Thread producerThread = new Thread(() -> {
            for (int i = 0; i < 5; i++) {
                data.produce(i);
                try {
                    Thread.sleep(1000);
                } catch (InterruptedException e) {
                    e.printStackTrace();
                }
            }
        });

        // Consumer thread
        Thread consumerThread = new Thread(() -> {
            for (int i = 0; i < 5; i++) {
                data.consume();
                try {
                    Thread.sleep(1000);
                } catch (InterruptedException e) {
                    e.printStackTrace();
                }
            }
        });

        // Start both threads
        producerThread.start();
        consumerThread.start();
    }
}

Q11. Write java code to demonstrate stored procedures with Callable Statement. 

package jdbcprog;
import java.sql.*;

public class Q12 {

    // Establishing connection
    public static Connection getConn() throws SQLException {
        Connection con=DriverManager.getConnection("jdbc:mysql://localhost:3306/studentdetails","root","root");
        return con;
    }

    // Inserting record using PreparedStatement
    public static void insertRecord(String name, int roll) {
        try {
            Connection con = getConn();
            PreparedStatement st=con.prepareStatement("insert into student values(?,?);");
            st.setString(1, name);
            st.setInt(2, roll);
            st.executeUpdate();
            System.out.println("Record inserted successfully...");
            printData();
        } catch(Exception e) {
            System.out.println(e);
        }
    }

    // Printing data from the student table
    public static void printData() {
        try {
            Connection con = getConn();
            Statement st = con.createStatement();
            ResultSet rs=st.executeQuery("select * from student;");
            while(rs.next()) {
                System.out.print(rs.getString(1)+" ");
                System.out.print(rs.getInt(2));
                System.out.println();
            }
        } catch(Exception e) {
            System.out.println(e);
        }
    }

    // Deleting record using PreparedStatement
    public static void deleteRecord(int roll) {
        try {
            Connection con = getConn();
            PreparedStatement st=con.prepareStatement("delete from student where rollno=?;");
            st.setInt(1, roll);
            st.executeUpdate();
            System.out.println("Record deleted successfully...");
            printData();
        } catch(Exception e) {
            System.out.println(e);
        }
    }

    // Example of using CallableStatement to call a stored procedure
    public static void callStoredProcedure(int roll) {
        try {
            Connection con = getConn();
            CallableStatement cst = con.prepareCall("{call delete_student(?)}");
            cst.setInt(1, roll);
            cst.executeUpdate();
            System.out.println("Stored procedure executed successfully...");
            printData();
        } catch(Exception e) {
            System.out.println(e);
        }
    }

    public static void main(String[] args) {
        try {
            Class.forName("com.mysql.cj.jdbc.Driver");
            
            // Inserting a record
            insertRecord("ashish", 10);
            
           
           
            // Deleting a record using a stored procedure and CallableStatement
            callStoredProcedure(10);
        } catch(Exception e) {
            System.out.println(e);
        }
    }
}

Q12. . Write java code to implement concept of Scrollable and updatable result sets.

package mypack;
import java.sql.*;
public class Q12 {

    public static void main(String[] args) {
        try {
            Class.forName("com.mysql.cj.jdbc.Driver");
            Connection connect = DriverManager.getConnection("jdbc:mysql://localhost:3306/studentdetails","root","root");
            Statement stmt = connect.createStatement(ResultSet.TYPE_SCROLL_SENSITIVE, ResultSet.CONCUR_UPDATABLE);

            ResultSet result = stmt.executeQuery("SELECT * from student");

            // Check if ResultSet is updatable
            //note : table must contain the primary key.
            if (!result.getMetaData().isReadOnly(1)) {
                while(result.next()) {
                    int no = result.getInt("registerNo");
                    String pass = result.getString("password");
                    System.out.println("Registration No: " + no + " Password: " + pass);
                }

                // Move to the last row
                result.last();

                // Update password
                result.updateString("password", "ashish");
                result.updateRow();

                // Insert new row
                result.moveToInsertRow();
                result.updateInt("registerNo", 43424);
                result.updateString("password", "abcd");
                result.insertRow();
                System.out.println("Operation done successfully");
            } else {
                System.out.println("ResultSet is not updatable");
            }

            result.close();
            stmt.close();
            connect.close();
        } catch(Exception e) {
            System.out.println(e);
        }
    }
}

Q13. Write java code to Making use of Database Metadata and ResultSetMetadata

package jdbcprog;
 
import java.sql.*;
 
public class Q13 {
   public static void main(String[] args) {
       Connection connection = null;
 
       try {
           // Establish connection
           connection = DriverManager.getConnection("jdbc:mysql://localhost:3306/studentdetails", "root", "root");
 
           // Get Database Metadata
           DatabaseMetaData dbMetaData = connection.getMetaData();
 
           // Display database information
           System.out.println("Database Product Name: " + dbMetaData.getDatabaseProductName());
           System.out.println("Database Product Version: " + dbMetaData.getDatabaseProductVersion());
           System.out.println("Driver Name: " + dbMetaData.getDriverName());
           System.out.println("Driver Version: " + dbMetaData.getDriverVersion());
 
           // Get ResultSet Metadata
           Statement statement = connection.createStatement();
           ResultSet resultSet = statement.executeQuery("SELECT * FROM student");
 
           ResultSetMetaData rsMetaData = resultSet.getMetaData();
 
           // Display ResultSet Metadata
           int columnCount = rsMetaData.getColumnCount();
           System.out.println("Number of columns: " + columnCount);
           System.out.println("Column Details:");
 
           for (int i = 1; i <= columnCount; i++) {
               System.out.println("Column " + i + " Name: " + rsMetaData.getColumnName(i));
               System.out.println("Column " + i + " Type: " + rsMetaData.getColumnTypeName(i));
               System.out.println("Column " + i + " Size: " + rsMetaData.getColumnDisplaySize(i));
           }
 
           // Process ResultSet
           System.out.println("Data from ResultSet:");
           while (resultSet.next()) {
               System.out.println("Name: " + resultSet.getString("name") + ", Roll: " + resultSet.getInt("rollno"));
           }
       } catch (SQLException e) {
           e.printStackTrace();
       } finally {
           // Close connection
           try {
               if (connection != null) {
                   connection.close();
               }
           } catch (SQLException e) {
               e.printStackTrace();
           }
       }
   }
}
